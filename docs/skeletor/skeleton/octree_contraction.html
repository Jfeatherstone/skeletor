231def _medianDistanceConnectionCriteria(boxPoints, neighborPoints, threshold):
232    """
233    Compute the median distances for each cell to the
234    plane defined by the cells's centroids, and the
235    normal vector between the two centroids.
236
237    Calculate the same quantity for the set of
238    points from merging the two cells.
239
240    If the latter quantity (times the threshold) is
241    less than the minimum of the two former quantities,
242    the two cells are neighbors.
243    """
244    # Calculate centroids
245    boxCentroid = np.mean(boxPoints, axis=0)
246    neighborCentroid = np.mean(neighborPoints, axis=0)
247    separation = neighborCentroid - boxCentroid 
248    combinedCentroid = boxCentroid + separation/2
249
250    # Find the plane in between the two centroids
251    planeNormal = combinedCentroid / np.sqrt(np.sum(combinedCentroid**2))
252
253    # Calculate d1, d2, and d12 (from the paper)
254    d1 = _calculateSquaredMedianDistance(boxPoints, boxCentroid, planeNormal)
255    d2 = _calculateSquaredMedianDistance(neighborPoints, neighborCentroid, planeNormal)
256    d12 = _calculateSquaredMedianDistance(np.concatenate((boxPoints, neighborPoints), axis=0), combinedCentroid, planeNormal)
257    
258    return d12*threshold <= min(d1, d2)
259

262def _calculateSquaredMedianDistance(points, planePoint, planeNormal):
263    """
264    Compute the squared median distance for points to a plane.
265    """
266
267    d = np.dot(planeNormal, planePoint)
268    planeDistance = (np.dot(planeNormal, np.transpose(points)) - d).T**2
269
270    return np.median(planeDistance)

858def contractPoints(points, referencePoints=None, pointMasses=None, attraction=1.0, contraction=0.5):
859    """
860    Perform Laplacian contraction on a set of points, potentially in relation to
861    a reference set of points (points that should attract other points but not
862    move themselves).
863    """
864    dim = np.shape(points)[-1]
865
866    if hasattr(referencePoints, '__iter__'):
867        allPoints = np.concatenate((points, referencePoints))
868
869        # We need a certain amount of points for the process to actually
870        # do anything; 5 is arbitrary
871        if len(allPoints < 5):
872            return points
873
874        # robust_laplacian's point cloud laplacian calculation uses the
875        # nearby nearby neighbors, and requires that you have at least
876        # n_neighbors points, which is equal to 30 by default.
877
878        # Compute the laplacian and mass matrix
879        L, M = robust_laplacian.point_cloud_laplacian(allPoints, n_neighbors=min(30, len(allPoints)-1))
880
881        # We should only have positive attraction towards reference
882        # points, and we should have slight negative attraction towards
883        # regular points (to avoid clumping)
884        if hasattr(pointMasses, '__iter__'):
885            pointRepulsion = attraction * pointMasses
886        else:
887            pointRepulsion = attraction * np.ones(len(points))
888
889        # Multiply the attraction of the reference points by a very large number so
890        # they don't move from their original positions much
891        referencePointAttraction = attraction * np.ones(len(referencePoints))*1e6
892        pointContraction = contraction * 1e3 * np.sqrt(np.mean(M.diagonal())) * np.ones(len(points))
893        referencePointContraction = contraction * 1e3 * np.sqrt(np.mean(M.diagonal())) * np.ones(len(referencePoints))
894
895        # Define weight matrices
896        WH = sparse.diags(np.concatenate((pointRepulsion, referencePointAttraction)))
897        WL = sparse.diags(np.concatenate((pointContraction, referencePointContraction)))  # I * laplacian_weighting
898
899    else:
900        allPoints = points
901        
902        # We need a certain amount of points for the process to actually
903        # do anything; 5 is arbitrary
904        if len(allPoints < 5):
905            return points
906
907        # Compute the laplacian and mass matrix
908        L, M = robust_laplacian.point_cloud_laplacian(allPoints, n_neighbors=min(30, len(points)-1))
909
910        attractionWeights = attraction * np.ones(M.shape[0])
911        # This is weighted by the sqrt of the mean of the mass matrix, not really sure why, but :/
912        contractionWeights = contraction * 1e3 * np.sqrt(np.mean(M.diagonal())) * np.ones(M.shape[0])
913
914        # Define weight matrices
915        WH = sparse.diags(attractionWeights)
916        WL = sparse.diags(contractionWeights)  # I * laplacian_weighting
917
918    A = sparse.vstack([L.dot(WL), WH]).tocsc()
919    b = np.vstack([np.zeros((allPoints.shape[0], 3)), WH.dot(allPoints)])
920
921    A_new = A.T @ A
922
923    # Solve each dimension separately
924    solvedAxes = [sla.spsolve(A_new, A.T @ b[:,i], permc_spec='COLAMD') for i in range(dim)]

926    # we will throw it away eventually)
927    if dim == 2:
928        solvedAxes += [list(points[:,2])]
929    ret = np.vstack(solvedAxes).T
930
931    if (np.isnan(ret)).all():
932        #logging.warn('Matrix is exactly singular. Stopping Contraction.')
933        ret = points
934
935    return ret[:len(points)]
